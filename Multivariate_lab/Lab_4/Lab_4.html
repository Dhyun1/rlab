Algorithm ,

  • Initialize \(\Psi^{(0)}\). for k = 1,2,
  • Calculate Uniqueness-rescaled covariance \(S^{*(k-1)}=\Psi^{-1/2(k-1)}S_n\Psi^{-1/2(k-1)}\).
  • Get eigenvalues \(\hat{\Lambda}^{(k-1)}\) and eigenvectors \(\hat{E}^{(k-1)}\).
  • Calculate \(L^{(k)}\) by \(L^{(k)}=\Psi^{1/2(k-1)}\hat{E}^{(k-1)}(I_m+\hat{\Lambda}^{(k-1)})^{1/2}\)
  • Calculate \(\Psi^{(k)}\) by \(\Psi^{(k)}=diag(S_n-L^{(k)}{L^{(k)}}^T)\).
  • Stop when it converges

psi <- (1-m/(2*p))*diag(1/diag(solve(cov)))

tol=10^(-5)             # For iteration
psi.old=diag(rep(0,5))  # For iteration
iter.num=1              # For iteration

,

while(norm(as.matrix(psi-psi.old),type="F")>tol){
  psi.old<-psi;
  S.star<-sqrt(solve(psi.old))%*%cov%*%sqrt(solve(psi.old))*(n-1)/n   ## Uniqueness-rescaled covariance 
  eig.res<-eigen(S.star);                                             ## Eigen decomposition
  E<-eig.res$vectors[,1:m]; Delta<-diag(eig.res$values[1:m]-1);       ## Ehat, Deltahat
  L<-sqrt(psi.old)%*%E%*%sqrt(Delta)                                  ## Calculate L
  psi<-diag(diag(cov*(n-1)/n-L%*%t(L)))                               ## Calculate Psi
  for(i in 1:p){
    if(diag(psi)[i]<0){
      diag(psi)[i]=0.005
    }                                                                 ## For Haywood case
  }
  iter.num<-iter.num+1
  # cat("iter.num=",iter.num,"psi=",psi[1,1],psi[2,2],psi[3,3],psi[4,4],psi[5,5],"\n")

}
