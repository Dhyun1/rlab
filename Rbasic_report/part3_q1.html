#a function that returns Season with input=Month.
Season<-function(month){
  s = month %/% 3
  if (!(month %in% c(1:12))){
    print("Not available number")
  }
  else if (s==0||s==4){
    return("Winter")
  } 
  else if (s==1){
    return("Spring")
  } 
  else if (s==2){
    return("Summer")
  } 
  else return("Fall")
}

#make df for printing
NO2<-data.spread2%>%
